const {MongoClient}= require('mongodb')
const url= "mongodb://127.0.0.1:27017";
const database='Student'
const client =new MongoClient(url)

async function insertData(){
    let result= await client.connect();
    let db=result.db(database);
    let collection=db.collection('stu');
    let doc = {name:'Rani', age:25};
    collection.insertOne(doc,
        function(err, result){
            if (err) throw err;
            console.log('Document inserted:', result.ops);
            
        });
    
}

insertData();

async function updateData(){
    let result= await client.connect();
    let db=result.db(database);
    let collection=db.collection('stu');
    let query = {name:'Rani'};
    let update = {$set:{name:'Rani', age:20}};
    let options = {upsert:true};

    collection. updateOne(query, update, options,
        function(err, result){
            if (err) throw err;
            console.log('Document updated:', result.ops);
            
        });
    
}
updateData();

async function deleteData(){
    let result= await client.connect();
    let db=result.db(database);
    let collection=db.collection('stu');
    let doc = {name:'Rani'};
    deleteResult = await collection.deleteOne(doc,
        function(err, result){
            if (err) throw err;
            console.log('Document deleted:', result.ops);
            
        });
    console.dir(deleteResult.deletedCount);
}
deleteData();
